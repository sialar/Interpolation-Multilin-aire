245 void Interpolation<T>::launchAIAlgo(bool debug)
246 {
247  auto start_time = chrono::steady_clock::now();
248 
249  // Initiatisation de m_curentNeighbours, m_path (séquence des points d'interpolation), argmax (le meilleur candidat)
250  m_curentNeighbours.clear();
251  m_path.clear();
252  MultiVariatePointPtr<T> argmax = getFirstMultivariatePoint();
253  MultiVariatePointPtr<T> old;
254  m_curentNeighbours.push_back(argmax);
255  int iteration = 0;
256 
257  // Si on atteint le nombre max d'itération, ou s'il n'y a plus de voisins possible
258  while (!m_curentNeighbours.empty() && iteration < m_maxIteration)
259  {
260  // On commence par calculer (si ce n'est pas déjà fait) les alpha des candidats (points voisin à path)
261  for (MultiVariatePointPtr<T> nu : m_curentNeighbours)
262  if (!nu->alphaAlreadyComputed())
263  computeLastAlphaNu(nu);
264 
265  // Pour suivre la progression de l'algo sur le terminal
266  if (debug)
267  {
268  cout << endl << endl;
269  displayPath();
270  displayCurentNeighbours();
271  }
272 
273  // Une fois tous les alpha calculés, on choisi le meilleur
274  argmax = maxElement(iteration);
275  // On l'ajoute à m_path (ensemble courant des points d'interpolation)
276  m_path.push_back(argmax);
277  addInterpolationPoint(getPoint(argmax));
278 
279  // Mise à jour de la liste des candidats et du numéro de l'itération
280  updateCurentNeighbours(argmax);
281  iteration++;

289 template <typename T>
290 void Interpolation<T>::computeLastAlphaNu(MultiVariatePointPtr<T> nu)
291 {
292  double basisFuncProd = 1.0;
293  vector<double> res = func(getPoint(nu));
294  m_nbEvals++;
295  for (MultiVariatePointPtr<T> l : m_path)
296  {
297  basisFuncProd = 1.0;
298 
299  for (int p=0; p<m_d; p++)
300  basisFuncProd *= basisFunction_1D((*l)(p),getPoint(nu)(p),p);
301  for (int k=0; k<m_n; k++)
302  res[k] -= l->getAlpha()[k] * basisFuncProd;
303  }
304  nu->setAlpha(res);
305 }

308 vector<double> Interpolation<T>::interpolation(MultiVariatePoint<double>& x, int end)
309 {
310  double l_prod = 1.0;
311  vector<double> sum(m_n,0.0);
312  for (int k=0; k<end; k++)
313  {
314  l_prod = 1.0;
315  for (int i=0; i<m_d; i++)

317  for (int i=0; i<m_n; i++)
318  sum[i] += (m_path[k]->getAlpha())[i] * l_prod;

399 template <typename T>
400 void Interpolation<T>::computeAIApproximationResults()
401 {
402  // Implémenter ici le type d'erreur qu'on veut calculer
403  vector<double> exactValue, approxValue, error;
404  vector<double> exactValuesNorm, errorsNorm;
405 
406  for (int i=0; i<m_nbTestPoints; i++)
407  {
408  exactValue = func(m_testPoints[i]);
409  approxValue = interpolation(m_testPoints[i]);
410  error = Utils::diff(exactValue,approxValue);
411 
412  m_exactValues.push_back(exactValue);
413  m_approxValues.push_back(approxValue);
414  m_errors.push_back(error);
415 
416  exactValuesNorm.push_back(Utils::norm(exactValue,2));
417  errorsNorm.push_back(Utils::norm(error,2));
418  }
419 
420  m_mseError = 0;
421  vector<double> relativeErrors;
422  double maxValue = *max_element(exactValuesNorm.begin(),exactValuesNorm.end());
423 
424  for (int i=0; i<m_nbTestPoints; i++)
425  {
426  relativeErrors.push_back(errorsNorm[i]*pow(10,5)/maxValue);
427  m_mseError += pow(errorsNorm[i]*pow(10,5),2)/m_nbTestPoints;
428  }
429 
430  m_infError = *max_element(relativeErrors.begin(), relativeErrors.end());
431  m_mseError = sqrt(m_mseError);
432 }
