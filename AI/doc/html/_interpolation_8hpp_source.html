347 template <typename T>
348 void Interpolation<T>::launchAIAlgo(bool debug)
349 {
350  // Début du calcul du temps d'execution
351  auto start_time = chrono::steady_clock::now();
352 
353  // Initiatisation de m_curentNeighbours, m_path (séquence des points d'interpolation), argmax (le meilleur candidat courant)
354  m_curentNeighbours.clear();
355  m_path.clear();
356  // Constuit le vecteur d'ordres correspondant au premier point d'interpolation
357  MultiVariatePointPtr<T> argmax = getFirstMultivariatePoint();
358  // Ajouter le premier point à la liste des voisins courant
359  m_curentNeighbours.push_back(argmax);
360  // Initilisation de la valeur de l'iteration courante
361  int iteration = 0;
362 
363  // On s'arrete si on atteint le nombre max d'itération, ou s'il n'y a plus de voisins possible
364  while (!m_curentNeighbours.empty() && iteration < m_maxIteration)
365  {
366  // On commence par calculer (si ce n'est pas déjà fait) les alpha des candidats (points voisins à path)
367  for (MultiVariatePointPtr<T> nu : m_curentNeighbours)
368  if (!nu->alphaAlreadyComputed())
369  computeLastAlphaNu(nu);
370 
371  // Pour suivre la progression de l'algo sur le terminal
372  if (debug)
373  {
374  cout << endl << endl;
375  displayPath();
376  displayCurentNeighbours();
377  }
378 
379  // Une fois tous les alpha calculés, on choisi le meilleur
380  argmax = maxElement(iteration);
381  // On ajoute le vecteur d'ordres correspondant dans m_path (ensemble courant des points d'interpolation)
382  m_path.push_back(argmax);
383  // On ajoute le point d'interpolation correspondant dans m_interpolationNodes et m_interpolationPoints
384  addInterpolationPoint(getPoint(argmax));

398 template <typename T>
399 void Interpolation<T>::computeLastAlphaNu(MultiVariatePointPtr<T> nu)
400 {
401  // alpha(nu) = f(y_{nu}) - I_{Lambda_{nu}}(f)(y_{nu})
402  // ou encore: alpha = f(y_{nu}) - \sum_{mu<nu} alpha(mu) * H_{mu}(y_{nu}) (rapport page 12)
403  double basisFuncProd = 1.0;
404  vector<double> res = func(getPoint(nu)); // f(y_{nu})
405  // Incrémentation du nombre de calcul de f éffectué par l'algo AI
406  m_nbEvals++;
407  // représente la somme sur les points d'interpolation courants
408  for (MultiVariatePointPtr<T> mu : m_path)
409  {
410  // Calcul de H, produit tensoriel (H_{mu}(f)(y_{nu})) des fonctions hiérarchiques au point d'interpolation y_{nu}
411  basisFuncProd = 1.0;
412  for (int p=0; p<m_d; p++)
413  basisFuncProd *= basisFunction_1D((*mu)(p),getPoint(nu)(p),p);
414 
415  for (int k=0; k<m_n; k++)
416  // correspond à alpha(mu) * H_{mu}(f)(y_{nu})
417  res[k] -= mu->getAlpha()[k] * basisFuncProd;
418  }
419  // Sauvegarder la valeur de l'erreur au point d'ordre nu
420  nu->setAlpha(res);
421 }

423 template <typename T>
424 vector<double> Interpolation<T>::interpolation(MultiVariatePoint<double>& x, int end)
425 {
426  // I_{Lambda_{n}}(f)(x) = \sum_{k<n} alpha(nu_k) * H_{nu_k}(x) (rapport page 12)
427  double l_prod = 1.0;
428  vector<double> sum(m_n,0.0);
429  for (int k=0; k<end; k++)
430  {
431  // Calcul de H_{nu_k}, produit tensoriel des fonctions hiérarchiques au point x
432  l_prod = 1.0;
433  for (int i=0; i<m_d; i++)
434  l_prod *= basisFunction_1D((*m_path[k])(i),x(i),i);
435  // Calucl de la somme des H_{nu_k} multipliés par les alpha des k premiers points
436  for (int i=0; i<m_n; i++)
437  sum[i] += (m_path[k]->getAlpha())[i] * l_prod;
438  }
439  return sum;
440 }

521 template <typename T>
522 void Interpolation<T>::computeAIApproximationResults()
523 {
524  // Calculer les erreurs d'interpolation (erreur relative absolue, erreur quadratique moyenne) sur
525  // la grille des points de test, m_testPoints
526  vector<double> exactValue, approxValue, error;
527  vector<double> exactValuesNorm, errorsNorm;
528 
529  for (int i=0; i<m_nbTestPoints; i++)
530  {
531  // valeur exacte de f au ième point de référence
532  exactValue = func(m_testPoints[i]);
533  // valeur approché de f au ième point de référence
534  approxValue = interpolation(m_testPoints[i]);
535  // erreur d'interpolation de f au ième point de référence
536  error = Utils::diff(exactValue,approxValue);
537 
538  m_exactValues.push_back(exactValue);
539  m_approxValues.push_back(approxValue);
540  m_errors.push_back(error);
541 
542  // Calcul des normes des valeurs exacte pour la recherche du sup d'entre eux
543  exactValuesNorm.push_back(Utils::norm(exactValue,2));
544  // Calcul des normes des erreurs
545  errorsNorm.push_back(Utils::norm(error,2));
546  }
547 
548  m_mseError = 0;
549  vector<double> relativeErrors;
550  // max des valeurs exacte, pour le calcul de l'erreur relative
551  double maxValue = *max_element(exactValuesNorm.begin(),exactValuesNorm.end());
552 
553  for (int i=0; i<m_nbTestPoints; i++)
554  {
555  // calcul et sauvegarde des erreurs relative sur chaque point de test
556  relativeErrors.push_back(errorsNorm[i]*pow(10,5)/maxValue);
557  // calcul de l'erreur quadratique moyenne
558  m_mseError += pow(errorsNorm[i]*pow(10,5),2)/m_nbTestPoints;
559  }
560 
561  m_infError = *max_element(relativeErrors.begin(), relativeErrors.end());
562  m_mseError = sqrt(m_mseError);
563 }
